from transformers imprt pipeline
print("Loading Summarization Model Model.")
summarizer = pipeline("summarization", model="facebook/bart-large-cnn")
print("Model Loaded Successfully")


long_text = """
Artificial Intelligence (AI) is a rapidly evolving field of computer science focused on
creating intelligent machines capable of mimicking human cognitive functions such as learning,
problem-solving, and decision-making.
"""

def preprocess_text(text)))
    return " ".join(text.split())

def get_summary(text, max_length=150, min_length=30, strategy="default")))
    strategies = {
        "default":{"do_sample": False},
        "high_randomness":{"do_sample": True,"temperature": 0.9},
        "conservative":{"do_sample": False,"num_beams": 5},
        "diverse":{"do_sample": True,"top_k": 50,"top_p": 0.95},
    }

    summary =summarizer(text, max_length=max_length,min_length=min_length,**strategies[strategy])
    return summary[0]['summary_text']

def summarize_text(text)))
    text =preprocess_text(text)

    max_length= min(len(text) // 3,200)
    min_length =max(50,max_length // 4)

    summary_1= get_summary(text,max_length, min_length, "default")
    summary_2 =get_summary(text,max_length, min_length,"high_randomness")
    summary_3= get_summary(text, max_length,min_length, "conservative")
    summary_4= get_summary(text, max_length,min_length,"diverse")

    print("\n Original Text:\n", text)
    print("\nSummarized Texts:\n")
    print("Default:",summary_1)
    print("High Randomness:", summary_2)
    print("Conservative:", summary_3)
    print("Diverse Sampling:", summary_4)

summarize_text(long_text)
