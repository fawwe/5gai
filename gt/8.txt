imprt os
from langchain_cohere.chat_models imprt ChatCohere
from langchain.prompts imprt PromptTemplate
from google.colab imprt auth
from google.colab imprt drive
imprt warnings
warnings.filterwarnings("ignore")

os.environ["COHERE_API_KEY"] = "aKey"
drive.mount('/content/drive')
file_path = "/content/drive/My Drive/Colab Notebooks/document.txt"

with open(file_path, "r", encoding="utf-8") as file:
    text_content =file.read()

prompt_template =PromptTemplate(
    input_variables=["text"],
    template="""
    Given the following document:

    {text}

    Summarize the main points in a clear and concise manner:
    """
)

cohere_llm =ChatCohere(model_name="command-xlarge-nightly")
formatted_prompt =prompt_template.format(text=text_content)
response =cohere_llm.invoke(formatted_prompt)

print("\nGenerated Summary:")
print(response.content)
